    let nextIndex = (currentIndex + 1) % sliderCount;
                        //(0 + 1) % 5 -> 1
                        //(1 + 1) % 5 -> 2
                        //(2 + 1) % 5 -> 3
                        //(3 + 1) % 5 -> 4
                        //(4 + 1) % 5 -> 0
                        //(0 + 1) % 5 -> 1
                        //(0 + 1) % 5 -> 2

    //slider.eq(currentindex).fadeOut();
    //slider[currentIndex]  //슬라이드를 배열식으로 해서 첫번째 인자를 가져오라.
    slider[currentIndex].style.opacity = "0";
    slider[nextIndex].style.opacity = "1"; 
    currentIndex = nextIndex;
}, 3000);

Slider Effect04

이미지 슬라이드 - 이미지 슬라이드(제이쿼리,자바스크립트)

  • jQuery
  • JavaScript
const sliderWrap = $(".slider_wrap");
const sliderImg = $(".slider_img");
const slider = $(".slider");
const sliderBtn = $(".slider-btn");

let currentIndex = 0;
let sliderCount = slider.length;
let sliderWidth = slider.width();
let sliderHeight = slider.height();
let sliderTotalWidth = sliderWidth * sliderCount; //슬라이드 이미지 개수가 바뀌어도 작동이되게

sliderImg.css({width: sliderTotalWidth, height: sliderHeight, marginLeft: -sliderWidth}); //marginleft를 안주면 왼쪽에는 이미지가 없으니까 슬라이드img 
$(".slider:last-child").prependTo(sliderImg);                                             //시작점을 옮겨줄라고  미리 왼쪽에 준비해둬야 자연스럽게 움직이기 때문에 줌
//오른쪽은 상관없지만 왼쪽에는 해놔야 이미지가 자연스럽게 움직임

function moveLeft(){
    //alert("왼쪽 버튼을 클릭")
    sliderImg.animate({left: + sliderWidth }, 400, function(){
        $(".slider:last-child").prependTo(sliderImg);
        sliderImg.css("left","");
    }); //기준점이 left로 되있어서 left만 줌
}
function moveRight(){
    //alert("오른쪽 버튼을 클릭")
    sliderImg.animate({left: - sliderWidth }, 400, function(){
        $(".slider:first-child").appendTo(sliderImg);
        sliderImg.css("left","");
    }); 
}

sliderBtn.find(".prev").click(function(){
    moveLeft();
});
sliderBtn.find(".next").click(function(){
    moveRight();
});

;

Slider Effect06

이미지 슬라이드 - 3단 슬라이드 만들기(제이쿼리,자바스크립트)

  • jQuery
  • JavaScript
const sliderWrap = $(".slider_wrap");
const sliderImg = $(".slider_img");
const slider = $(".slider");
const sliderBtn = $(".slider-btn");

let sliderCount = slider.length;
//let sliderWidth = slider.width();
let sliderWidth = slider.outerWidth(); //padding 값을 포함하지 않은 순수 width값 부를때
let sliderShowNumber = 3;
let currentIndex = 0;

let sliderCopy = $(".slider:lt("+sliderShowNumber+")").clone(); //12345뒤에 123이 있음
//콘솔로그로 확인하면 slider123이 나타남
sliderImg.append(sliderCopy);

//alert(sliderWidth); ->380(padding을 줬기 때문에 400이였던 width값이 380로 나옴)
function moveLeft(){  //처음이미지일때
    if(currentIndex == 0){  //0과 같다면 5를 대입시켜줘라
        currentIndex = sliderCount; //5=sliderCount
        sliderImg.css("margin-left", sliderWidth * -currentIndex);
    }
    currentIndex--; //클릭할때마다 이미지가 바뀌도록
    sliderImg.animate({"margin-left": -sliderWidth * currentIndex},400);
}
function moveRight(){  //마지막 이미지일때
    if(currentIndex == sliderCount){  
        currentIndex = 0;
        sliderImg.css("margin-left",currentIndex);
    }
    currentIndex++; //클릭할때마다 이미지가 바뀌도록
    sliderImg.stop().animate({"margin-left": -sliderWidth * currentIndex},400); //0일떄 0 1일떄 400 2일때 800 ...
}       //margin주고 -를 주면 왼쪽으로 간다고 생각하면 됨

//버튼 만들기
sliderBtn.on("click","a",function(){
    if($(this).hasClass("prev")){
        moveLeft();               
    } else {
        moveRight();
    }
});

;

    dotIndex += "";  //+를 해주는 이유는 이미지 개수가 증가해도 이미지 개수만큼 나올수 있도록 해주기 위해 
});

//     dotIndex += ""+name+"";
// });

//닷 버튼 클릭하기
sliderDot.on("click","a",function(e){
    e.preventDefault();
    gotoSlider($(this).index()); //현재 숫자가 어떤이미지인지 알고 싶으려면 index메소드이용
    //alert($(this).index());

sliderBtn.on("click","a",function(e){   //a태그 포함하여 클릭 이벤트 추가

    //}
