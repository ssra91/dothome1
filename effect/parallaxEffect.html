Parallax03 - jQuery

scrollTop과 offsetTop을 이용하여 패럴랙스 스크롤링 효과로 이질감을 느끼도록 화면을 완성하였습니다.

let scrollTop = $(window).scrollTop();  //시간차가 생기게 함수 밖에 안에 따로 scrollTop을 준다 100일 때 90

function scroll(){
    const newScrollTop = $(window).scrollTop();
    const different = newScrollTop - scrollTop; //오차가 나게 하기 위해서 빼준다.

    console.log("scrollTop :" + scrollTop);
    console.log("newScrollTop :" + newScrollTop);
    console.log("different :" + different);

    gsap.to(".content-item-imgWrap",{duration: .3, skewY: different * 0.2, ease: Power1.ease});
    gsap.to(".content-item-num",{duration: .6, x: different * 3, ease: Power1.ease});
    gsap.to(".content-item-desc",{duration: .6, skewX: different * 0.2, ease: Power4.ease});

    // $("content-item-imgWrap").css({"transform":"skewY(" + different + "deg)"});

    scrollTop = newScrollTop;  //초기화 현상
    requestAnimationFrame(scroll);   //1초에 60번 돌리는 효과 마우스가 움직일 때마다 (different값)
}
scroll();
